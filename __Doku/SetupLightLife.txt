************ Setup Samba
sudo nano /etc/samba/smb.conf

[global]
   workgroup = WORKGROUP
	realm = raspberrypi  ##new
	netbios name = raspberrypi ##new
	
'*** einkommentieren
   security = user

   '*** add the end of samba.conf add
	[lightlife]
	path = /lightlife
	writeable = yes
	guest ok = no
	

'*** after all config
'*** create samba users
sudo smbpasswd -a lightlife
'***[Enter 2 times the password]

sudo /etc/init.d/samba restart
